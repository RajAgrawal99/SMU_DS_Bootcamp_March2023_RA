inertia = []
silhouettes = []
k = list(range(2, 12))

for i in k:
    print(f"Processing K={i}")
    k_model = KMeans(n_clusters=i, random_state=1)
    k_model.fit(df_scale) # fit on the scaled data
    inertia.append(k_model.inertia_)
    
    preds = k_model.predict(df_scale)
    silhouette_avg = silhouette_score(df_scale, preds)
    silhouettes.append(silhouette_avg)
